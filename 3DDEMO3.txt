IF _ASM THEN @INITACLS:CLEARASMGRP$="ASM_3D"LOADER$="3DDEMO3"EXEC "_ASMLOAD"IF !RESULT THEN ?"Error: Missing _ASMLOAD"END@INITCLEAR'POINTER OFFSETSP1=1/4096P2=P1*2P4=P1*4P8=P1*8P128=P1*128AX=&H000AY=&H000M=0GOSUB @3DINIT@MAINGOSUB @3DTESTVSYNC 1IF !(BUTTON() AND 16)GOTO @MAINGOSUB @3DOFFEND@3DINIT 'ENABLE 3D RENDER 1RETURN @3DTEST B=BUTTON() IF B AND 1 THEN AY=AY-P128 IF B AND 2 THEN AY=AY+P128 IF B AND 4 THEN AX=AX+P128 IF B AND 8 THEN AX=AX-P128 B=BTRIG() IF B AND 256 THEN M=M-(M>0) IF B AND 512 THEN M=M+(M<5)  ' Projection matrix ' (Based on GBATEK example) RMTX 0,0,SQR(3),-11/9,-20/9  ' Translate z=-4 RMTX 0,3,0,0,-4 'ROTATE around Y = in XZ RMTX 0,4,0,COS(AY),SIN(AY)'ROTATE around X = in YZ  RMTX 0,4,1,COS(AX),SIN(AX) RPOLY -1,M  LOCATE 0,0 ?"X:";DEG(AX) ?"Y:";DEG(AY) 'SWAP BUFFER RENDER 1RETURN@3DOFF RENDER 0END
